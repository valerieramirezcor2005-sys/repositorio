function actualizarIndice(){

let uxVal=Number(ux.value)

let freqVal=Number(freq.value)

let timeVal=Number(time.value)

indice.innerText=((uxVal+freqVal+timeVal)/3).toFixed(1)

uxData[5]=uxVal

freqData[5]=freqVal

timeData[5]=timeVal

behaviorChart.update()

}

document.querySelectorAll("input").forEach(e=>e.addEventListener("input",actualizarIndice))

// PORTAFOLIO KPIs

function actualizarPortafolio(){

let b = Number(invBogota.value)

let m = Number(invMedellin.value)

let c = Number(invCali.value)

let total = b + m + c

totalInv.innerText = total

b_inv.innerText = b + "%"

m_inv.innerText = m + "%"

c_inv.innerText = c + "%"

let eng = ((b*8.2)+(m*7.5)+(c*6.8))/total

engagementTotal.innerText = eng.toFixed(1)

let riesgoScore = ((b*2)+(m*1)+(c*3))/total

let riesgoTxt = "Medio"

if(riesgoScore < 1.8) riesgoTxt = "Bajo"

else if(riesgoScore > 2.3) riesgoTxt = "Alto"

riesgoTotal.innerText = riesgoTxt

let roiCalc = (eng * 2.2) - riesgoScore

roi.innerText = roiCalc.toFixed(1) + "%"

let max = Math.max(b,m,c)

if(max === b) canal.innerText = "Bogotá"

else if(max === m) canal.innerText = "Medellín"

else canal.innerText = "Cali"

actualizarGraficaPortafolio(b,m,c)

}

document.querySelectorAll("#portafolio input").forEach(e=>{

e.addEventListener("input", actualizarPortafolio)

})
